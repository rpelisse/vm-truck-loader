22  package org.redhat.vmtruckloader.vmware;
23  
24  import java.net.URL;
25  
26  import javax.inject.Inject;
27  
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  /**
32   * Very simple implementation of the VMWareContext.
33   *
34   * @author <a href="mailto:duncan.doyle@redhat.com">Duncan Doyle</a>
35   * @author Romain Pelisse - romain@redhat.com
36   */
37  public class VMWareContext {
38  
39  	@Inject
40  	private static final Logger LOGGER  = LoggerFactory.getLogger(VMWareContext.class);
41  		
42  	private final URL vcenterUrl;
43  	private String vcenterUsername;
44  	private String vcenterPassword;
45  
46  	public VMWareContext(String username, String password,
47  			URL vCenterURL2) {
48  		vcenterUrl = vCenterURL2;
49  		vcenterUsername = username;
50  		vcenterPassword = password;
51  		
52  		if (LOGGER.isDebugEnabled()) {
53  			LOGGER.debug("Configured " + this.getClass().getSimpleName() + " with url '" + vcenterUrl + "'.");
54  		}
55  	}
56  
57  	public URL getVCenterURL() {
58  		return vcenterUrl;
59  	}
60  
61  	public String getUsername() {
62  		return vcenterUsername;
63  	}
64  
65  	public String getPassword() {
66  		return vcenterPassword;
67  	}
68  }
