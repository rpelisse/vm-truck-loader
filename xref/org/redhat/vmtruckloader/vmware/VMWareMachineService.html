48  /**
49   * <p>VMWare implementation of the {@link MachineService} responsible for {@link Machine}
50   * related semantics (list, start, stop, etc.).</p>
51   *
52   * @author Duncan Doyle - duncan.doyle@redhat.com
53   * @author Romain Pelisse - belaran@redhat.com
54   */
55  public class VMWareMachineService implements MachineService { //NOPMD, class is long but pretty simple, refactoring not needed
56  
57  	@Inject
58  	private VMWareTemplate vmwareTemplate;
59  	
60  	@Override
61  	public void setConnectionData(String username, String password, URL vCenterUrl) {
62  		vmwareTemplate.setCredential(username, password, vCenterUrl);
63  	}
64  
65  	public Collection<VirtualMachine> getMachines() {
66  		return vmwareTemplate.execute(new GetMachinesActionCallback());
67  	}
68  
69  	@Override
70  	public VirtualMachine getMachine(final String machineName) {
71  		return vmwareTemplate.execute(new GetMachineActionCallback(machineName));
72  	}
73  
74  	@Override
75  	public synchronized void startMachine(String machineName) { //NOPMD, block level sync is not an option here
76  		vmwareTemplate.execute(new StartMachineActionCallback(machineName, true));
77  	}
78  
79  	@Override
80  	public synchronized void stopMachine(String machineName) { //NOPMD, block level sync is not an option here
81  		stopMachine(machineName,true);
82  	}
83  
84  	public synchronized void stopMachine(String machineName, boolean async) { //NOPMD, block level sync is not an option here
85  		vmwareTemplate.execute(new StopMachineActionCallback(machineName, async));
86  	}
87  
88  	@Override
89  	public void deleteMachine(String machineName) {
90  		vmwareTemplate.execute(new DeleteMachineActionCallback(machineName));
91  	}
92  
93  	@Override
94  	public MachineSpecification cloneMachine(String sourceMachineName, MachineSpecification spec) {
95  		return vmwareTemplate.execute(new CloneMachineCallback(sourceMachineName, spec));
96  	}
97  
98  	@Override
99  	public Collection<Task> getJobs() {
100 		final VMWareActionCallback<Collection<Task>> callback = new AbstractVMWareActionCallback<Collection<Task>>() {
101 
102 			@Override
103 			public Collection<Task> doInVmware(ServiceInstance serviceInstance) {
104 				Task[] recentTasks = serviceInstance.getTaskManager().getRecentTasks();
105 				List<Task> tasks = new ArrayList<Task>(recentTasks.length);
106 				for ( Task task: recentTasks ) {
107 					tasks.add(task);
108 				}
109 				return tasks;
110 			}
111 
112 		};
113 		return vmwareTemplate.execute(callback);
114 	}
115 
116 	@Override
117 	public void deploy(String netName, String vmName) {
118 		vmwareTemplate.execute(new MoveVirtualMachineToNetworkCallback(netName, vmName));
119 	}
120 
121 	@Override
122 	public void editMachine(MachineSpecification machineCreate) {
123 		vmwareTemplate.execute(new EditMachineCallback(machineCreate));
124 	}
