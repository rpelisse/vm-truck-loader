24  import java.util.ArrayList;
25  import java.util.Collection;
26  import java.util.List;
27  
28  import org.redhat.vmtruckloader.vmware.VMWareManagedObjectUtils;

30  import com.vmware.vim25.mo.Folder;
31  import com.vmware.vim25.mo.InventoryNavigator;
32  import com.vmware.vim25.mo.ManagedEntity;

37  /**
38   * Simple VMWare action class that executes a finds all VMs.
39   * <p/>
40   * Note that these action classes are not designed to be thread safe.
41   *  
42   * @author <a href="mailto:duncan.doyle@redhat.com">Duncan Doyle</a>
43   * @author Romain Pelisse - romain@redhat.com
44   */
45  public class GetMachinesActionCallback extends AbstractVMWareActionCallback<Collection<VirtualMachine>> {
46  	
47  	@Override
48  	public Collection<VirtualMachine> doInVmware(ServiceInstance serviceInstance) {
49  		return getVms(serviceInstance);
50  	}
51  	
52  	private List<VirtualMachine> getVms(ServiceInstance serviceInstance) {
53  		final List<VirtualMachine> virtualMachines = new ArrayList<VirtualMachine>();
54  
55  		final Folder rootFolder = serviceInstance.getRootFolder();
56  		final InventoryNavigator inventoryNavigator = new InventoryNavigator(rootFolder);
57  
58  		for (ManagedEntity managedEntity : VMWareManagedObjectUtils.getManagedEntities(inventoryNavigator, VirtualMachine.class.getName())) {
59  			virtualMachines.add((VirtualMachine) managedEntity);
60  		}
61  		return virtualMachines;
62  	}

64  }
